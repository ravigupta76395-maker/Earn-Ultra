import dbConnect from '../../../lib/dbConnect'; import User from '../../../models/User'; export default async function handler(req, res) { if (req.method !== 'GET') return res.status(405).end(); const { telegramId } = req.query; await dbConnect(); const top = await User.find({ referralCount: { $gt: 0 } }) .sort({ referralCount: -1 }) .limit(50) .select('telegramId firstName username referralCount totalEarned'); let myRank = null; let myData = null; if (telegramId) { const me = await User.findOne({ telegramId: String(telegramId) }); if (me) { const rank = await User.countDocuments({ referralCount: { $gt: me.referralCount } }); myRank = rank + 1; myData = { telegramId: me.telegramId, firstName: me.firstName, username: me.username, referralCount: me.referralCount, totalEarned: me.totalEarned, rank: myRank, }; } } return res.status(200).json({ leaderboard: top.map((u, i) => ({ rank: i + 1, telegramId: u.telegramId, name: u.firstName || u.username || 'User', referralCount: u.referralCount, totalEarned: u.totalEarned, })), myData, }); }
